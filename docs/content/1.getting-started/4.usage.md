## Storefront API

```html [app/pages/example.vue]
<script setup lang="ts">
const storefront = useStorefront()

const { data: products } = useAsyncData('products', () => {
    const gql = `#graphql
        query FetchProducts($first: Int) {
            products(first: $first) {
                edges {
                    node {
                        id
                        title
                        description
                    }
                }
            }
        }
    `

    return storefront.request(gql, {
        variables: {
            first: 9,
        },
    })
})
</script>

<template>
    <div>
        <h1>Products</h1>
        <pre>{{ products.value.edges }}</pre>
    </div>
</template>
```
